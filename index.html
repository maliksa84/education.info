import java.util.InputMismatchException; import java.util.Random; import java.util.Scanner; public class VisualZooRoulette { public static void main(String[] args) { System.out.println("----------------------------------------"); System.out.println(" Welcome to ZOO ROULETTE (Visual) "); System.out.println("----------------------------------------"); int currentPoints = 5000; // Animals and their multipliers as shown in the image String[] animals = {"Monkey", "Rabbit", "Pigeon", "Peacock", "Lion", "Shark", "Eagle"}; double[] multipliers = {2.0, 6.0, 8.0, 12.0, 12.0, 24.0, 8.0}; Scanner scanner = new Scanner(System.in); Random random = new Random(); while (currentPoints > 0) { System.out.println("\n----------------------------------------"); System.out.println("Current Points: " + currentPoints); System.out.println("Place your bet on an animal:"); // Displaying the animals in a grid-like fashion System.out.println("1. Monkey (x" + multipliers[0] + ") | 2. Rabbit (x" + multipliers[1] + ") | 3. Pigeon (x" + multipliers[2] + ")"); System.out.println("4. Peacock (x" + multipliers[3] + ") | 5. Lion (x" + multipliers[4] + ") | 6. Shark (x" + multipliers[5] + ")"); System.out.println("7. Eagle (x" + multipliers[6] + ")"); int userChoice = 0; int betAmount = 0; try { System.out.print("Enter your choice (1-" + animals.length + "): "); userChoice = scanner.nextInt(); if (userChoice < 1 || userChoice > animals.length) { System.out.println("Invalid choice. Please select a valid number."); continue; } System.out.print("Enter your bet amount (min 20): "); betAmount = scanner.nextInt(); if (betAmount < 20) { System.out.println("Bet must be 20 or more."); continue; } if (betAmount > currentPoints) { System.out.println("You can't bet more points than you have. Your points: " + currentPoints); continue; } } catch (InputMismatchException e) { System.out.println("Invalid input! Please enter a number."); scanner.nextLine(); // Clear the invalid input from scanner continue; } scanner.nextLine(); // Consume the newline character String userAnimal = animals[userChoice - 1]; double multiplier = multipliers[userChoice - 1]; System.out.println("\nYou chose: " + userAnimal); System.out.println("You bet: " + betAmount + " points."); System.out.println("Roulette wheel is spinning..."); try { Thread.sleep(2000); // Simulate spinning } catch (InterruptedException e) { // Ignore } int winningIndex = random.nextInt(animals.length); String winningAnimal = animals[winningIndex]; System.out.println("The winning animal is: " + winningAnimal); System.out.println("----------------------------------------"); if (userAnimal.equalsIgnoreCase(winningAnimal)) { int winnings = (int) (betAmount * multiplier); currentPoints += winnings; System.out.println("Congratulations! You WON " + winnings + " points!"); } else { currentPoints -= betAmount; System.out.println("Oh no! You lost " + betAmount + " points. Better luck next time!"); } System.out.println("Your new points: " + currentPoints); if (currentPoints <= 0) { System.out.println("\n----------------------------------------"); System.out.println("Game Over! You have no more points."); System.out.println("----------------------------------------"); break; } System.out.print("\nDo you want to play again? (yes/no): "); String playAgain = scanner.nextLine(); if (!playAgain.equalsIgnoreCase("yes")) { System.out.println("Thank you for playing! Final Points: " + currentPoints); break; } } scanner.close(); } }
